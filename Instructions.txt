
== Control Plane Node Setup ==
# install container runtime to manage containers 
cat <<EOF | sudo tee /etc/modules-load.d/k8s.conf 
overlay 
br_netfilter 
EOF 

# to immediately activate overlay & br_netfilter 
sudo modprobe overlay 
sudo modprobe br_netfilter 

# configure systcl for network settings 
cat <<EOF | sudo tee /etc/sysctl.d/k8s.conf 
net.bridge.bridge-nf-call-iptables = 1 
net.bridge.bridge-nf-call-ip6tables = 1 
net.ipv4.ip_forward = 1 
EOF 

# activate the sysctl network settings 
sudo sysctl --system 

# install containerd, update apt package first then install containerd 
sudo apt-get update 
sudo apt-get install -y containerd 

# configure containerd.  
sudo mkdir -p /etc/containerd 
sudo containerd config default | sudo tee /etc/containerd/config.toml 

# this cmd basically finish updating the configuration above. 
sudo sed -i 's/SystemCgroup = false/SystemCgroup = true/' /etc/containerd/config.toml 

# restart and enable containerd 
sudo systemctl restart containerd 
sudo systemctl enable containerd 

# Disable swap. The kubelet requires swap disabled to ensure predictable resource mgt & performance. The sudo sed cmd is to make it persistent across reboots.  
sudo swapoff -a 
sudo sed -i '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab 

# add the k8s repo 
sudo apt-get install -y apt-transport-https ca-certificates curl gpg 

# this dir is for the next command 
sudo mkdir -p -m 755 /etc/apt/keyrings 

# this will dload k8s sign key and register with apt 
curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.29/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg 

# adding this command so that apt knows where to look for kubelet,kubeadm & kube-controller. 
echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.29/deb/ /" | sudo tee /etc/apt/sources.list.d/kubernetes.list 

# then update the apt package 
sudo apt-get update 
# install kubelet, kubeadm, kubectl (select no if service asked to be restarted for now. 
sudo apt-get install -y kubelet kubeadm kubectl  

# this cmd will prevent accidental upgrade, which is best practice for maintaining a stable version.  
sudo apt-mark hold kubelet kubeadm kubectl 


== NOTE: Do not do this part in the worker node ==  
# configure a single node cluster. The --pod-network-cidr flag specifies the IP add range for pod network as this is required by the CNI. Once you applied the Calico YAML on the control plane, it automatically deploys "DaemonSets" to every node that joins the cluster. It will configure itself on the worker node automatically so you not need to install Calico. Take note of the kubeadm join command as you would need that to run in the worker node to join the cluster.  
sudo kubeadm init --pod-network-cidr=10.244.0.0/16 

#configure kubectl then create dir then copy the cluster admin config file created by kubeadm to the standard location. 
mkdir -p $HOME/.kube 
sudo cp -i /etc/kubernetes/admin.conf $HOME/.kube/config 

# change the ownership so that non root user can securely interact w/ the cluster using the kubectl tool & the user can now run kubectl commands to manage the cluster w/out needing sudo. 
sudo chown $(id -u):$(id -g) $HOME/.kube/config 

# remove the controlplane taint as by default the controlplane is tainted so that regular app pods is prevented to be scheduled here. For a single node cluster, the taint is removed. 
kubectl taint nodes --all node-role.kubernetes.io/control-plane- 

# install CNI plugin, we'll use Calico. The cluster is not fully functional until a container network interface (CNI) plugin is installed. Without it the pods cannot communicate with each other and core DNS will not start correctly. 
kubectl apply -f https://raw.githubusercontent.com/projectcalico/calico/v3.28.0/manifests/calico.yaml
